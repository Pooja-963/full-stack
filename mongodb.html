const MongoClient = require('mongodb').MongoClient; // Connection URL const url = 'mongodb://localhost:27017'; // Database Name const dbName = 'mydatabase'; // Create a new MongoClient const client = new MongoClient(url, { useUnifiedTopology: true }); // Connect to the MongoDB server client.connect((err) => { if (err) { return; } console.log('Connected to MongoDB server'); const db = client.db(dbName); // Perform CRUD operations here // Close the MongoDB connection client.close(); }) ; CRUD OPERTION: // Get the collection const collection = db.collection('users'); // Create a new document const user = { name: 'John Doe', email: 'john@example.com', } ; // Insert the document collection.insertOne(user, (err, result) => { if (err) { console.error('Error inserting document:', err); console.error('Error connecting to MongoDB:', err); } console.log('Document inserted:', result.insertedId); }) ; READ (FIND): // Get the collection const collection = db.collection('users'); // Find documents collection.find({}).toArray((err, documents) => { if (err) { console.error('Error retrieving documents:', err); return; } console.log('Documents:', documents); }) ; UPDATE: // Get the collection const collection = db.collection('users'); // Update a document const filter = { name: 'John Doe' }; const update = { $set: { email: 'newemail@example.com' } }; collection.updateOne(filter, update, (err, result) => { return; console.error('Error updating document:', err); return; } console.log('Document updated:', result.modifiedCount); }) ; DELETE: // Get the collection const collection = db.collection('users'); // Delete a document const filter = { name: 'John Doe' }; collection.deleteOne(filter, (err, result) => { if (err) { console.error('Error deleting document:', err); return; } console.log('Document deleted:', result.deletedCount); })
